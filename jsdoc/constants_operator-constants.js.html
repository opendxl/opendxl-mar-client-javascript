/**
 * @module OperatorConstants
 * @description Constants that describe the `operator` to use within a
 * `condition`.
 *
 * The following statement:
 *
 * ```js
 * {
 *   or: [{
 *     and: [{
 *       name: 'HostInfo',
 *       output: 'ip_address',
 *       op: 'EQUALS',
 *       value: '192.168.1.1'
 *     }]
 *   }]
 * }
 * ```
 *
 * Can be rewritten to use `OperatorConstants` as follows:
 *
 * ```js
 * {
 *   or: [{
 *     and: [{
 *       name: 'HostInfo',
 *       output: 'ip_address',
 *       op: OperatorConstants.EQUALS,
 *       value: '192.168.1.1'
 *     }]
 *   }]
 * }
 * ```
 */

'use strict'
